
自定义左边框红色背景
```
<?xml version="1.0" encoding="utf-8"?>
<layer-list xmlns:android="http://schemas.android.com/apk/res/android">
    <item>
        <shape>
            <!--    边框颜色    -->
            <solid android:color="#9f383d"/>
        </shape>
    </item>
    <!-- 左边边框3dp -->
    <item
        android:left="3dp">
        <shape>
            <solid android:color="#ffffff"/>
        </shape>
    </item>
</layer-list>

```
