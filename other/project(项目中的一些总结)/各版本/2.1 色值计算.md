            float temp = 255 * i * 1.0f / 100f;
            int alpha = Math.round(temp);
            String hexStr = Integer.toHexString(alpha);
            if (hexStr.length() < 2)
                hexStr = "0" + hexStr;
            System.out.println(i + "%, " + hexStr.toUpperCase());

      }
