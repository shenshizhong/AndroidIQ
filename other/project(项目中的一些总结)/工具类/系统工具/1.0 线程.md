class MyThread : Thread(){
    override fun run() {
        super.run()
        println("我是子线程")
    }
}

/** 通过实现 Runnable 实现子线程*/
class MyThread2 : Runnable{
    override fun run() {
        println("我是子线程2:" + Thread.currentThread().name)
    }
}
