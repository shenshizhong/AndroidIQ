	public void bubbleSort(int sums[]) {
		int temp = 0;
		for (int i = 0; i < sums.length - 1; i++) {
		  boolean flag = true;
			for (int j = 0; j < sums.length - 1 - i; j++) {
				if (sums[j] > sums[j + 1]) {
					temp = sums[j];
					sums[j] = sums[j + 1];
					sums[j + 1] = temp;
					flag = false
				}
			}
			if(flag){//已经是从小到大，就直接跳过进行下一轮，降低复杂度
      	break;
      }
		}
	}
